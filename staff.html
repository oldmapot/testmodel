function doCalc() { const isTicket = document.getElementById('ticketMode').checked; let newPts = 0, redeem = 0; if (isTicket) { const buy = Number(document.getElementById('buyTicket').value) || 0; newPts = Math.floor(buy * config.ticketRate); } else { const total = Number(document.getElementById('totalSpend').value) || 0; const ticket = Number(document.getElementById('ticketSpend').value) || 0; const base = total - ticket; const max = base >= config.minSpend ? Math.min(currentPoints, Math.floor(base * config.maxRate)) : 0; document.getElementById('max-hint').innerText = max; redeem = Math.min(Number(document.getElementById('redeemAmount').value) || 0, max); document.getElementById('redeemAmount').value = redeem; // 🔥 這行才是你原本正確的：用會員等級倍率！ newPts = Math.floor((base - redeem) * memberLevelRate); } // 防 NaN newPts = isNaN(newPts) ? 0 : newPts; const afterPts = (currentPoints || 0) - redeem + newPts; document.getElementById('new-pts').innerText = newPts; document.getElementById('after-pts').innerText = afterPts; }
